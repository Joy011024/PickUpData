-- =============================================
--说明:IP地址转换成格式化数字IP
--时间:2011-12-25 15:30 HCH
-- =============================================

CREATE FUNCTION [dbo].[f_IP2Int]
(
    @ip CHAR(15)
)
RETURNS BIGINT
AS
BEGIN
    DECLARE @re BIGINT   
    SET @re = 0   
    SELECT @re = @re+LEFT(@ip, CHARINDEX('.', @ip+'.')-1)*ID, @ip = STUFF(@ip, 1, CHARINDEX('.', @ip+'.'), '') 
    FROM (
             SELECT ID = CAST(16777216 AS BIGINT) 
             UNION ALL   SELECT 65536 
             UNION ALL   SELECT 256 
             UNION ALL   SELECT 1
         )A 
    RETURN(@re)
END
GO

-- =============================================
--说明:数字IP转换成格式化  IP地址
--时间:2011-12-25 15:30 HCH
-- =============================================
CREATE FUNCTION [dbo].[f_Int2IP]
(
    @IP BIGINT
)
RETURNS VARCHAR(15)
AS
BEGIN
    DECLARE @re VARCHAR(16)   
    SET @re = ''   
    SELECT @re = @re+'.'+CAST(@IP/ID AS VARCHAR), @IP = @IP%ID 
    FROM (
             SELECT ID = CAST(16777216 AS BIGINT) 
             UNION ALL   SELECT 65536 
             UNION ALL   SELECT 256 
             UNION ALL   SELECT 1
         )a 
    RETURN(STUFF(@re, 1, 1, ''))
END

-- Procedure
-- =============================================	
-- 时间: 2011.11.23 HCH
-- 用途: SQL SERVER工作计划使用,用来清除超过30天的日志
-- =============================================
CREATE PROCEDURE [dbo].[_JOB_DeleteOverdueLog]
 AS
BEGIN
	SET NOCOUNT ON;
--日志记录数量(达到一定数量才启动过期删除)
	DECLARE @Count BIGINT;
--过期时间
	DECLARE  @OverTime DATETIME;
	SELECT @OverTime=DATEADD(DAY,-30,GETDATE())
--删除过期角色封停日志
	SELECT @Count=COUNT(1) FROM T_LockRoleLog WITH(NOLOCK)
	IF @Count>10000000
	DELETE FROM T_LockRoleLog WHERE F_LockDeleteTime<@OverTime
--删除过期用户封停日志
	SELECT @Count=COUNT(1) FROM T_LockUserLog WITH(NOLOCK)
	IF @Count>10000000
	DELETE FROM T_LockUserLog WHERE F_LockDeleteTime<@OverTime
--删除过期角色登录退出游戏日志
	SELECT @Count=COUNT(1) FROM T_RoleEnterExitLog WITH(NOLOCK)
	IF @Count>100000000
	DELETE FROM T_RoleEnterExitLog WHERE F_ExitTime<@OverTime
--删除过期交易金钱日志
--	SELECT @Count=COUNT(1) FROM T_TradeMoneyLog WITH(NOLOCK)
--	IF @Count>100000000
--	DELETE FROM T_TradeMoneyLog WHERE F_TradeTime<@OverTime
--删除过期GM借用帐号日志
--	SELECT @Count=COUNT(1) FROM T_UserGMUsedLog WITH(NOLOCK)
--	IF @Count>10000000
--	DELETE FROM T_UserGMUsedLog WHERE F_BackTime<@OverTime
--删除过期用户登录退出日志表
	DECLARE @i INT;--遍历索引
	DECLARE @tablename VARCHAR(30);--表名
	--查询是否存在日志临时表
	IF(EXISTS (SELECT 1 FROM tempdb..sysobjects WHERE id=object_id('tempdb..#TempLog')))        
	DROP TABLE #TempLog
	--得到用户登录退出日志表
	SELECT (ROW_NUMBER() OVER(order by id asc))AS rownum,name INTO #TempLog FROM sysobjects WITH(NOLOCK)  WHERE name LIKE 'T_UserEnterExitLog%' and name<>'T_UserEnterExitLog' AND OBJECTPROPERTY(id, 'IsUserTable') = 1
	--遍历得到的日志表
	SELECT @i=MAX(rownum) FROM #TempLog
	WHILE @i>0
	BEGIN
	SELECT @tablename=name FROM  #TempLog WHERE rownum=@i
	--超期日志表将删除
	IF (LEN(@tablename)>20 AND CAST(RIGHT(@tablename,8) AS smalldatetime) <@OverTime)
	begin EXEC('DROP TABLE '+@tablename+'') end
	else  begin SELECT @i=0 end
	SELECT @i=@i-1
	END
END

-- PROCEDURE
------------------------------------
--项目名称：数据库日志
--说明：用户登录退出日志
--时间：2011-12-2 10:35:59 HCH
------------------------------------
CREATE PROCEDURE [dbo].[_LOG_UserStatis]
@UserID INT,
@UserName NVARCHAR(50)
 AS 
BEGIN
declare @LoginCount int,@CTNLoginCount int,@MaxCTNLoginCount int,@MaxLoginInterval int,@LastLoginInterval int,@ActiveTime datetime
	BEGIN TRY
--检测是否存在
	SELECT @LoginCount=F_LoginCount,@CTNLoginCount=F_CTNLoginCount,@MaxCTNLoginCount=F_MaxCTNLoginCount,@MaxLoginInterval=F_MaxLoginInterval,@LastLoginInterval=F_LastLoginInterval,@ActiveTime=F_ActiveTime FROM T_UserStatis with(nolock) WHERE F_UserID=@UserID
	IF @CTNLoginCount is null
	BEGIN
		INSERT INTO T_UserStatis
			(F_UserID, F_UserName,F_LoginTimes,F_LoginCount, F_CTNLoginCount, F_MaxCTNLoginCount,F_MaxLoginInterval, F_LastLoginInterval, F_FirstLoginTime, F_ActiveTime)
		VALUES	(@UserID,@UserName ,1,1, 1, 1,1, 1, GETDATE(), GETDATE())
	END
	ELSE
	BEGIN
		select @LastLoginInterval=DATEDIFF(D,@ActiveTime,GETDATE())--距离上次登录天数
		if @LastLoginInterval=1 --连续登录天数
			select @CTNLoginCount=@CTNLoginCount+1
		if @CTNLoginCount>@MaxCTNLoginCount --最大连续登录天数
			select @MaxCTNLoginCount=@CTNLoginCount
		if @LastLoginInterval>0 --登录天数
			select @LoginCount=@LoginCount+1
		if @LastLoginInterval<=0--距离登录天数 最小为1天.为了保持回归状态一天
		select @LastLoginInterval=1
		if @LastLoginInterval>@MaxLoginInterval --最大距离登录天数
			select @MaxLoginInterval=@LastLoginInterval
		--更新数据
		UPDATE    T_UserStatis
		SET F_LoginTimes=F_LoginTimes+1,F_LoginCount = @LoginCount, F_CTNLoginCount = @CTNLoginCount, F_MaxCTNLoginCount = @MaxCTNLoginCount, F_MaxLoginInterval=@MaxLoginInterval,F_LastLoginInterval = @LastLoginInterval, F_ActiveTime = getdate()
		WHERE (F_UserID = @UserID)
	END


	END TRY
	BEGIN CATCH
	INSERT INTO T_Error(F_Note) VALUES ('_LOG_UserStatis '+ERROR_MESSAGE())
	RETURN (1)
	END CATCH
END
GO

-- PROCEDURE
------------------------------------
--项目名称：数据库日志
--说明：用户登录退出日志
--时间：2011-12-2 10:35:59 HCH
------------------------------------
CREATE PROCEDURE [dbo].[_LOG_UserEnterExitLog]
@UserLogedID BIGINT,
@UserID INT,
@UserName NCHAR(50),
@LoginNGSID SMALLINT,
@ZoneID SMALLINT,
@LoginIP CHAR(32),
@LoginType TINYINT,
@LoginTime DATETIME,
@ExitTime DATETIME
 AS 
BEGIN
	BEGIN TRY
	DECLARE @TABLENAME VARCHAR(30),@TableTime DATETIME
--判断表日期
	IF @ExitTime is NOT NULL
	BEGIN
		SELECT @TableTime=@ExitTime
		if Datediff(n,@LoginTime,@ExitTime)<0 SELECT @ExitTime=@LoginTime
	END
	ELSE
	SELECT @TableTime=@LoginTime
--检测是否存在当前日期的日志表
	SELECT @TABLENAME='T_UserEnterExitLog'+ convert(VARCHAR(8), @TableTime,112)
	IF NOT EXISTS (SELECT 1 FROM sysobjects WITH(NOLOCK)  WHERE id = OBJECT_ID(@TABLENAME) AND OBJECTPROPERTY(id, 'IsUserTable') = 1) 
	BEGIN
		EXEC('CREATE TABLE [dbo].['+@TABLENAME+'](
		[F_LogID] [BIGINT] IDENTITY(1,1) NOT NULL,
		[F_UserLogedID] [BIGINT] NULL,
		[F_UserID] [INT] NULL,
		[F_UserName] [NCHAR](50) NULL,
		[F_LoginNGSID] [SMALLINT] NULL,
		[F_ZoneID] [SMALLINT] NULL,
		[F_LoginIP] [NCHAR](32) NULL,
		[F_LoginType] [TINYINT] NULL,
		[F_TimeCount] [INT] NULL CONSTRAINT [DF_'+@TABLENAME+'_F_TimeCount]  DEFAULT ((0)),
		[F_LoginTime] [SMALLDATETIME] NULL,
		[F_ExitTime] [SMALLDATETIME] NULL,
	 CONSTRAINT [PK_'+@TABLENAME+'] PRIMARY KEY CLUSTERED 
	([F_LogID] ASC
	)WITH (PAD_INDEX  = OFF, STATISTICS_NORECOMPUTE  = OFF, IGNORE_DUP_KEY = OFF, ALLOW_ROW_LOCKS  = ON, ALLOW_PAGE_LOCKS  = ON) ON [PRIMARY]
	) ON [PRIMARY]
')
		EXEC('CREATE NONCLUSTERED INDEX [IX_'+@TABLENAME+'_UserLogedID] ON [dbo].['+@TABLENAME+'] 
			(
				[F_UserLogedID] ASC
			)WITH (PAD_INDEX  = OFF, STATISTICS_NORECOMPUTE  = OFF, SORT_IN_TEMPDB = OFF, IGNORE_DUP_KEY = OFF, DROP_EXISTING = OFF, ONLINE = OFF, ALLOW_ROW_LOCKS  = ON, ALLOW_PAGE_LOCKS  = ON) ON [PRIMARY]')
	END
--根据退出时间是否为空判断登录退出
	IF(@ExitTime is  NULL )
	BEGIN
	--新增日志数据
		EXEC('INSERT INTO ['+@TABLENAME+'](
		F_UserLogedID,F_UserID,F_UserName,F_LoginNGSID,F_ZoneID,F_LoginIP,F_LoginType,F_TimeCount,F_LoginTime,F_ExitTime
		)VALUES(
		'+@UserLogedID+','+@UserID+','''+@UserName+''','+@LoginNGSID+','+@ZoneID+','''+@LoginIP+''','+@LoginType+',0,'''+@LoginTime+''',NULL)')
	END
	ELSE
	BEGIN
		EXEC('SELECT top 1 * FROM ['+@TABLENAME+'] WHERE F_UserLogedID='+@UserLogedID+'' )
		IF(@@Rowcount=0)
		--新增日志数据
			EXEC('INSERT INTO ['+@TABLENAME+'](
			F_UserLogedID,F_UserID,F_UserName,F_LoginNGSID,F_ZoneID,F_LoginIP,F_LoginType,F_TimeCount,F_LoginTime,F_ExitTime
			)VALUES(
			'+@UserLogedID+','+@UserID+','''+@UserName+''','+@LoginNGSID+','+@ZoneID+','''+@LoginIP+''','+@LoginType+',Datediff(n,convert(VARCHAR(10), '''+@ExitTime+''',120),'''+@ExitTime+'''),convert(VARCHAR(10), '''+@ExitTime+''',120),'''+@ExitTime+''')')
		ELSE
		 --更新日志数据
			EXEC('UPDATE ['+@TABLENAME+'] SET F_TimeCount=Datediff(n,F_LoginTime,'''+@ExitTime+'''),F_ExitTime='''+@ExitTime+''' WHERE F_UserLogedID='+@UserLogedID+' ')
	END
	EXEC _LOG_UserStatis @UserID,@UserName
	END TRY
	BEGIN CATCH
	INSERT INTO T_Error(F_Note) VALUES ('_LOG_UserEnterExitLog '+ERROR_MESSAGE())
	RETURN (1)
	END CATCH
END
